void compute(Change change) {
  if (change == Change.course) {
    _go.clear();
    var newCourses = _courses.getCodes();
    for (var course in newCourses) {
      if (!_dropped.contains(course)) {
        _go.add(course);
      }
    }
    _dropped.removeWhere((course) => !newCourses.contains(course));
  }
}
