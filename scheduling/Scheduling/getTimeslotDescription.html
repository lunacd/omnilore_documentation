String getTimeslotDescription(int timeIndex) {
  if (timeIndex < 0 || timeIndex > 19) {
    throw const InvalidArgument(message: 'Invalid time index');
  }
  String result = '';
  if (timeIndex < 10) {
    result += '1 & 3';
  } else {
    result += '2 & 4';
  }
  switch (((timeIndex % 10) / 2).floor()) {
    case 0:
      result += ' Mon';
      break;
    case 1:
      result += ' Tue';
      break;
    case 2:
      result += ' Wed';
      break;
    case 3:
      result += ' Thu';
      break;
    case 4:
      result += ' Fri';
      break;
  }
  if (timeIndex % 10 % 2 == 0) {
    result += ' AM';
  } else {
    result += ' PM';
  }
  return result;
}
